# Simple noteApp

## 1 - create a config.env file

- NODE_ENV= (development or production)
- PORT = 1234
- DATABASE = (mongodb url)
- DATABASE_PASSWORD = (mongodb password)

- JWT_SECRET = App
- JWT_EXPIRES_IN = 10m
- JWT_COOKIE_EXPIRES_IN = 60

## run these command

### npm i

### npm run (start or dev)
